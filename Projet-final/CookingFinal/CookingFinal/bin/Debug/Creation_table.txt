CREATE TABLE `cooking`.`client` ( `pseudo` VARCHAR(20) NOT NULL, `motDePasse` VARCHAR(20) NOT NULL, `prenomClient` VARCHAR(20) NOT NULL, `nomClient` VARCHAR(20) NOT NULL, `numeroTelClient` VARCHAR(20) NULL, `creditCook` INT NOT NULL, `CdR` BOOL NOT NULL, PRIMARY KEY (`pseudo`) ); 
CREATE TABLE `cooking`.`commande` (`numeroCommande` INT NOT NULL, `dateCommande` DATE NOT NULL,`prixTotal` INT NOT NULL,`pseudo` VARCHAR(20) NOT NULL, PRIMARY KEY (`numeroCommande`), INDEX `F_comm1_idx` (`pseudo` ASC), CONSTRAINT `pseudo` FOREIGN KEY (`pseudo`) REFERENCES `cooking`.`client` (`pseudo`) ON DELETE CASCADE ON UPDATE NO ACTION ); 
CREATE TABLE `cooking`.`fournisseur` ( `nomFournisseur` VARCHAR(20) NOT NULL, `numeroTelFournisseur` VARCHAR(20) NOT NULL, PRIMARY KEY (`nomFournisseur`) ); 
CREATE TABLE `cooking`.`recette`(`numeroRecette` INT NOT NULL, `nomRecette` VARCHAR(30) NOT NULL, `typeRecette` VARCHAR(20)  NOT NULL, `descriptif` VARCHAR(150)  NULL, `prixVente` INT NOT NULL, `remunerationCdR` INT NOT NULL, `compteur` INT NOT NULL, `pseudo` VARCHAR(20) NOT NULL, PRIMARY KEY (`numeroRecette`), INDEX `F_comm1_idx` (`pseudo` ASC), FOREIGN KEY (`pseudo`) REFERENCES `cooking`.`client` (`pseudo`) ON DELETE CASCADE ON UPDATE NO ACTION );     
CREATE TABLE `cooking`.`contient` (`numeroCommande` INT NOT NULL, `numeroRecette` INT NOT NULL, `quantiteRecette` INT NOT NULL, PRIMARY KEY (`numeroCommande`, `numeroRecette`), INDEX `F_cont1_idx` (`numeroCommande` ASC), CONSTRAINT `numeroCommande` FOREIGN KEY (`numeroCommande`) REFERENCES `cooking`.`commande` (`numeroCommande`) ON DELETE CASCADE ON UPDATE NO ACTION , INDEX `F_cont2_idx` (`numeroRecette` ASC), CONSTRAINT `numeroRecette` FOREIGN KEY (`numeroRecette`) REFERENCES `cooking`.`recette` (`numeroRecette`) ON DELETE CASCADE ON UPDATE NO ACTION ); 
CREATE TABLE `cooking`.`produit` (`nomProduit` VARCHAR(20) NOT NULL, `categorie` VARCHAR(30) NOT NULL, `uniteQuantite` INT NOT NULL, `stockActuel` INT NOT NULL, `stockMinimal` INT NOT NULL, `stockMaximal` INT NOT NULL, `nomFournisseur` VARCHAR(20) NOT NULL, PRIMARY KEY (`nomProduit`), INDEX `F_prod1_idx` (`nomFournisseur` ASC), FOREIGN KEY (`nomFournisseur`) REFERENCES `cooking`.`fournisseur` (`nomFournisseur`) ON DELETE CASCADE ON UPDATE NO ACTION );            
CREATE TABLE `cooking`.`utilise` (`nomProduit` VARCHAR(20) NOT NULL, `numeroRecette` INT NOT NULL, `quantiteProduits` INT NOT NULL, PRIMARY KEY (`nomProduit`, `numeroRecette`), INDEX `F_util1_idx` (`nomProduit` ASC),	CONSTRAINT `nomProduit` FOREIGN KEY (`nomProduit`) REFERENCES `cooking`.`produit` (`nomProduit`) ON DELETE CASCADE ON UPDATE NO ACTION, INDEX `F_util2_idx` (`numeroRecette` ASC), FOREIGN KEY (`numeroRecette`) REFERENCES `cooking`.`recette` (`numeroRecette`) ON DELETE CASCADE ON UPDATE NO ACTION );
